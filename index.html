      // Example 1
  var chord1 = Chord("chord-ex1", {
    label: "F#M",
    base_fret: 0,
    scale: 1,
    notes: [
      { 'string': 0, 'finger': "T", 'fret': 2, 'tonic': true, 'class': "group1" },
      { 'string': 1, 'open': true },
      { 'string': 2, 'finger': "2", 'fret': 4, 'class': "group1" },
      { 'string': 3, 'finger': "3", 'fret': 2 },
      { 'string': 4, 'muted': true },
      { 'string': 5, 'finger': "2", 'fret': 2, 'class': "group1" }
    ]}
  );
  chord1.addNotes([
    { 'string': 2, 'finger': "2", 'fret': 5, 'class': "group1" }
  ], "red");
